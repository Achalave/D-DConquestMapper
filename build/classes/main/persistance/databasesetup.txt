CREATE TABLE STRUCTURES
(ID     INT     PRIMARY KEY     NOT NULL,
 NAME   TEXT    NOT NULL,
 X_LOC  INT     NOT NULL,
 Y_LOC  INT     NOT NULL,
 SIZE   INT     NOT NULL)

CREATE TABLE STRUCTURE_REALATIONS
(ID1    INT     NOT NULL,
 ID2    INT     NOT NULL,
 DISTANCE   INT     NOT NULL,
 PRIMARY KEY (ID1, ID2),
 FOREIGN KEY (ID1) REFERENCES STRUCTURES(ID),
 FOREIGN KEY (ID2) REFERENCES STRUCTURES(ID))


CREATE TABLE ITEMS
(ID     INT     PRIMARY KEY     NOT NULL,
 NAME   TEXT    NOT NULL)

//Stores the relation between items and people
CREATE TABLE ITEM_PERSON_RELATIONS
(IID    INT     NOT NULL,
 PID    INT     NOT NULL,
 NOTE   TEXT,
 PRIMARY KEY (IID, PID),
 FOREIGN KEY (IID) REFERENCES ITEM(ID),
 FOREIGN KEY (PID) REFERENCES PERSON(ID))

CREATE TABLE PERSONS
(ID     INT     PRIMARY KEY     NOT NULL,
 NAME   TEXT    NOT NULL,
 AGE    INT,
 SIZE   TEXT)

//Stores the locations that a person has been seen
CREATE TABLE PERSON_SEEN_LOCATIONS
(ID     INT     NOT NULL,
 X_LOC  INT     NOT NULL,
 Y_LOC  INT     NOT NULL)


CREATE TABLE EVENTS
(ID     INT     PRIMARY KEY     NOT NULL,
 NAME   TEXT    NOT NULL,
 DESCRIPTION    TEXT,
 X_LOC  INT     NOT NULL,
 Y_LOC  INT     NOT NULL)

//Stores the event relations to structures
CREATE TABLE EVENT_STRUCTURE_REALATIONS
(EID    INT     NOT NULL,
 SID    INT     NOT NULL,
 DISTANCE   INT     NOT NULL,
 PRIMARY KEY (EID, SID),
 FOREIGN KEY (EID) REFERENCES EVENT(ID),
 FOREIGN KEY (SID) REFERENCES STRUCTURES(ID))
